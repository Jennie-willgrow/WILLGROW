async function createReportsForDate(targetDateStr, teacherId) {

  try {

    var parts=targetDateStr.split('-');

    var targetDate=new Date(parseInt(parts[0]),parseInt(parts[1])-1,parseInt(parts[2]));

    var existFilter='select=title&date='+_eq(targetDateStr);

    if(teacherId) existFilter+='&teacher_id='+_eq(teacherId);

    var existRows = await sbGet('daily_reports', existFilter);

    var existingSet={};

    existRows.forEach(function(r){

      var base=(r.title||'').replace(/\s*\(\d+\/\d+\)$/,'').trim();

      if(base) existingSet[base]=true;

    });

    var srcRows=null;

    for(var w=1;w<=4;w++){

      var srcDay=new Date(targetDate); srcDay.setDate(targetDate.getDate()-w*7);

      var y=srcDay.getFullYear(),mo=String(srcDay.getMonth()+1).padStart(2,'0'),da=String(srcDay.getDate()).padStart(2,'0');

      var srcStr=y+'-'+mo+'-'+da;

      var fp='select=id,title,class_id,teacher_id,book,todays_lesson,homework&date='+_eq(srcStr);

      if(teacherId) fp+='&teacher_id='+_eq(teacherId);

      var attempt = await sbGet('daily_reports', fp);

      if(!attempt.length) continue;

      if(!attempt.some(function(r){ return r.todays_lesson&&r.todays_lesson.trim(); })) continue;

      srcRows=attempt; break;

    }

    if(!srcRows) return { success:false, error:'전주 데이터 없음 (최대 4주 소급)' };

    var m=targetDate.getMonth()+1, d=targetDate.getDate(), created=0;

    for(var i=0;i<srcRows.length;i++){

      var page=srcRows[i];

      var base=(page.title||'').replace(/\s*\(\d+\/\d+\)$/,'').trim();

      var newTitle=base+' ('+m+'/'+d+')';

      if(existingSet[base]) continue;

      var result=await sbInsert('daily_reports',{

        title:newTitle, date:targetDateStr,

        last_hw:page.homework||'', todays_lesson:'', homework:''

    }

}

function parseConsultDate(raw){

  if(!raw) return { dateStr:"", timeStr:"" };

  var dateStr = raw.split("T")[0];

  var timeStr = "";

  if(raw.indexOf("T") > -1){

    var tp = raw.split("T")[1].replace(/[+Z][\s\S]*$/,"").split(":");

    var h = parseInt(tp[0]||0), m = parseInt(tp[1]||0);

    var ampm = h >= 12 ? "PM" : "AM";

    var h12 = h > 12 ? h-12 : (h===0 ? 12 : h);

    timeStr = h12 + ":" + String(m).padStart(2,"0") + " " + ampm;

  }

  return { dateStr:dateStr, timeStr:timeStr };

}

function renderConsultCal(students){

  students.sort(function(a,b){ return a.consultDate.localeCompare(b.consultDate); });

  var year = consultCalMonth.year, month = consultCalMonth.month;

  var MONTHS = ["January","February","March","April","May","June","July","August","September","October","November","December"];

  var MSHORT = ["Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec"];

  var DAYS   = ["Sun","Mon","Tue","Wed","Thu","Fri","Sat"];

  var now = new Date(Date.now() + 9*60*60*1000);

  var todayY = now.getUTCFullYear(), todayM = now.getUTCMonth(), todayD = now.getUTCDate();

  var firstDow = new Date(year, month, 1).getDay();

  var daysInMonth = new Date(year, month+1, 0).getDate();

  // Build appt days map for this month

  var apptDays = {};

  students.forEach(function(s){

    var p = parseConsultDate(s.consultDate);

    if(!p.dateStr) return;

    var parts = p.dateStr.split("-");

    var sy=parseInt(parts[0]), sm=parseInt(parts[1])-1, sd=parseInt(parts[2]);

    if(sy===year && sm===month){ if(!apptDays[sd]) apptDays[sd]=[]; apptDays[sd].push(s); }

  });

  // Calendar header

  var html = "<div class='cal-header'>";

  html += "<button class='cal-nav-btn' onclick='changeConsultMonth(-1)'>‹</button>";

  html += "<div class='cal-month-lbl'>" + MONTHS[month] + " " + year + "</div>";

  html += "<button class='cal-nav-btn' onclick='changeConsultMonth(1)'>›</button>";

  html += "</div>";

  // Day-of-week headers

  html += "<div class='cal-grid'>";

  ["Sun","Mon","Tue","Wed","Thu","Fri","Sat"].forEach(function(d){ html += "<div class='cal-day-hdr'>"+d+"</div>"; });

  // Empty leading cells

  for(var i=0; i<firstDow; i++) html += "<div class='cal-cell'></div>";

  // Day cells

  for(var d=1; d<=daysInMonth; d++){

    var isToday = (year===todayY && month===todayM && d===todayD);

    var hasAppt = !!apptDays[d];

    var cls = "cal-cell" + (isToday?" cal-today":"") + (hasAppt?" cal-has-appt cal-clickable":"");

    var ds = year+"-"+String(month+1).padStart(2,"0")+"-"+String(d).padStart(2,"0");

    html += "<div class='"+cls+"'" + (hasAppt?" onclick='scrollToApptDay(\""+ds+"\")'":"") + ">";

    html += "<span class='cal-day-num'>"+d+"</span>";

    if(hasAppt) html += "<span class='cal-dot'>"+apptDays[d].length+"</span>";

    html += "</div>";

  }

  html += "</div>";

  // Appointment list

  html += "<div class='appt-list'>";

  if(!students.length){

    html += "<div class='empty'>No upcoming consultations.</div>";

  } else {

    var prevDate = "";

    students.forEach(function(s){

      var p = parseConsultDate(s.consultDate);

      if(!p.dateStr) return;

      if(p.dateStr !== prevDate){

        var dp = p.dateStr.split("-");

        var dn = new Date(parseInt(dp[0]), parseInt(dp[1])-1, parseInt(dp[2]));

        html += "<div class='appt-date-hdr' id='appt-"+p.dateStr+"'>";

        html += MSHORT[dn.getMonth()]+" "+dn.getDate()+", "+dn.getFullYear()+" ("+DAYS[dn.getDay()]+")";

        html += "</div>";

        prevDate = p.dateStr;

      }

      html += "<div class='appt-card'>";

      html += "<div class='appt-name'>"+s.name+(s.grade?"<span class='appt-grade'>"+s.grade+"</span>":"")+"</div>";

      if(p.timeStr) html += "<div class='appt-time'>🕐 "+p.timeStr+"</div>";

      if(s.parentContact) html += "<div class='appt-phone'>📞 <a href='tel:"+s.parentContact+"'>"+s.parentContact+"</a></div>";

      html += "</div>";

    });

  }

  html += "</div>";

  document.getElementById("consultCalContainer").innerHTML = html;

}

function _compressImage(file, maxPx, quality, callback){

  if(!file.type || !file.type.match(/^image\//)){

    var r=new FileReader();

    r.onload=function(e){ callback(e.target.result, file.type, file.name); };

    r.readAsDataURL(file); return;

  }

  var img=new Image();

  var url=URL.createObjectURL(file);

  img.onload=function(){

    URL.revokeObjectURL(url);

    var w=img.width, h=img.height;

    if(Math.max(w,h)>maxPx){ var s=maxPx/Math.max(w,h); w=Math.round(w*s); h=Math.round(h*s); }

    var c=document.createElement("canvas"); c.width=w; c.height=h;

    c.getContext("2d").drawImage(img,0,0,w,h);

    var newName=file.name.replace(/\.[^.]+$/,".jpg");

    callback(c.toDataURL("image/jpeg",quality), "image/jpeg", newName);

  };

  img.onerror=function(){

    URL.revokeObjectURL(url);

    var r=new FileReader();

    r.onload=function(e){ callback(e.target.result, file.type, file.name); };

    r.readAsDataURL(file);

  };

  img.src=url;

}
